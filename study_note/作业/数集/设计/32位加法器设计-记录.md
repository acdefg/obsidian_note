```verilog
module adder_cla_4bit #(parameter WIDTH = 4)(
	input [WIDTH-1:0] a,b,
	input ci,
	output wire [WIDTH-1:0] s,
	output co
);

wire [WIDTH-1:0] p,q;
wire [WIDTH-1:0] carry;

assign p = a & b;
assign q = a | b;
assign carry[0] = p[0] | (q[0] & ci);
assign carry[1] = p[1] | (p[0] & q[1]) | (&q[1:0] & ci);
assign carry[2] = p[2] | (p[1] & q[2]) | (p[0] & q[1] & q[2]) +
				  ((&q[2:0]) & ci);
assign carry[3] = p[3] | (p[2] & q[3]) | (p[1] & q[2] & q[3]) +
				  (p[0] & (&q[3:1])) | ((&q[3:0]) & ci);
				  
assign s[0] = a[0] ^ b[0] ^ ci;
assign s[3:1] = a[3:1] ^ b[3:1] ^ carry[2:0];
assign co = carry[3];
endmodule

```
